This is for hopping in the x and y directions. The integrals all have the same sign,
hence the Hamiltonian becomes.
Using the Fourier transformation
$c_i = \frac{1}{\sqrt{N}} \sum_k e^{i \bm k \cdot \bm R_i} c_{\bm k}$
\begin{align}
    H &= t \sum_{R, \delta} c_{i+\delta}^\dagger c_i \\\nonumber
    &= \frac{t}{N} \sum_{R, \delta} \sum_{q, k} e^{-i \bm q \cdot (\bm R_i + \bm \delta)} e^{i \bm k \cdot \bm R_i} c_{\bm q}^\dagger c_{\bm k} \\\nonumber
    &= \frac{t}{N} \sum_\delta \sum_{q, k} \delta_{qk} e^{-i \bm q \cdot \bm \delta} c_q^\dagger c_k \\\nonumber
    &= t \sum_\delta \sum_k e^{-i q \cdot \delta} c_k^\dagger c_k \\\nonumber
    &= t\sum_k \left[e^{-ik_x a} + e^{-ik_y a} + e^{ik_x a} + e^{ik_y a}\right]c_k^\dagger c_k \\\nonumber
    &= 2t \sum_k \left[\cos(k_x a) + cos(k_y a)\right]c_k^\dagger c_k
\end{align}

The diagonal terms do not depend on the sign so we get.
\begin{align}
    H &= t \sum_\delta \sum_k e^{-ik \cdot \delta} c_k^\dagger c_k \\\nonumber
    &= t \sum_k \left[e^{-ia(k_x+k_y)}+e^{-ia(k_x-k_y)}+e^{-ia(-k_x+k_y)} + e^{-ia(-k_x-k_y)}\right] c_k^\dagger c_k \\\nonumber
    &= t \sum_k \left[2\cos(k_xa + k_ya) + 2\cos(k_xa - k_ya)\right] c_k^\dagger c_k \\\nonumber
    &= 4t \sum_k \cos(k_x a)\cos(k_y a) c_k^\dagger c_k \\\nonumber
\end{align}
for the off diagonal terms we get a minus sign for $a(-\hat x, \hat y)$ and $a(\hat x, -\hat y)$.
\begin{align}
    H &= t \sum_\delta \sum_k e^{-ik \cdot \delta} c_k^\dagger c_k \\\nonumber
    &= t \sum_k \left[e^{-ia(k_x+k_y)}-e^{-ia(k_x-k_y)}-e^{-ia(-k_x+k_y)} + e^{-ia(-k_x-k_y)}\right] c_k^\dagger c_k \\\nonumber
    &= t \sum_k \left[(e^{-iak_x} - e^{iak_x})(e^{-iak_y}-e^{iak_y})\right] c_k^\dagger c_k \\\nonumber
    &= t \sum_k \left[(2i\sin(k_x a))(2i\sin(k_y a))\right] c_k^\dagger c_k \\\nonumber
    &= -4t \sum_k \sin(k_x a)\sin(k_y a) c_k^\dagger c_k \\\nonumber
\end{align}

\section*{Single Layer Hamiltonian}
Our matrix elements can be represented by the following functions
\begin{align}
    \varepsilon_{xy} &= -2t_\pi \left[\cos(k_x a) + \cos(k_y a)\right] - 4t_{2xy} \cos(k_x a)\cos(k_y a) \\\nonumber
    \varepsilon_{yz} &= -2t_\pi \cos(k_x a) - 2t_\delta \cos(k_y a) - 4t_{2yzzx} \cos(k_x a)\cos(k_y a) \\\nonumber
    \varepsilon_{zx} &= -2t_\delta \cos(k_x a) - 2t_\pi \cos(k_y a) - 4t_{2yzzx} \cos(k_x a)\cos(k_y a) \\\nonumber
    g &= +4t_{g} \sin(k_x a)\sin(k_y a)
\end{align}
and the matrix is in our basis of $yz, xz, xy$
\begin{gather}
    H_0 = \begin{pmatrix}
        \varepsilon_{yz} & g & 0 \\
        g & \varepsilon_{xz} & 0 \\
        0 & 0 & \varepsilon_{xy}
    \end{pmatrix}
\end{gather}
